  <script>
    const fileInput = document.getElementById('fileInput');
    const dropZone = document.getElementById('dropZone');
    const globalDropOverlay = document.getElementById('globalDropOverlay');
    const fileInfo = document.getElementById('fileInfo');
    const fileNameEl = document.getElementById('fileNameEl');
    const previewContainer = document.getElementById('previewContainer');
    const previewEmpty = document.getElementById('previewEmpty');
    const previewWrapper = document.getElementById('previewWrapper');
    const previewStage = document.getElementById('previewStage');
    const previewImage = document.getElementById('previewImage');
    const overlayCanvas = document.getElementById('overlayCanvas');
    const cutLinesLayer = document.getElementById('cutLinesLayer');
    const octx = overlayCanvas.getContext('2d');
    const manualList = document.getElementById('manualList');
    const zoomSlider = document.getElementById('zoomSlider');
    const zoomValue = document.getElementById('zoomValue');
    const btnProcess = document.getElementById('btnProcess');
    const btnDownloadAll = document.getElementById('btnDownloadAll');
    const btnReset = document.getElementById('btnReset');
    const results = document.getElementById('results');
    const btnChangeFile = document.getElementById('btnChangeFile');
    const geminiApiKey = document.getElementById('geminiApiKey');
    const sourceText = document.getElementById('sourceText');
    const translatedText = document.getElementById('translatedText');
    const btnTranslateSelected = document.getElementById('btnTranslateSelected');
    const btnSaveApiKey = document.getElementById('btnSaveApiKey');
    const translateInfo = document.getElementById('translateInfo');

    let originalImage = null;
    let objectUrl = null;
    let splitPoints = [];
    let selectedLineIndex = -1;
    let draggingIndex = -1;
    let isDraggingLine = false;
    let resultImages = [];
    let zoomPercent = 100;
    let dragLastClientY = 0;
    let dragScrollTimer = null;
    let dragScrollDir = 0;
    let cutPolicy = 'auto';
    let windowDragCounter = 0;
    const STORAGE_GEMINI_KEY = 'imageCutterGeminiApiKey';

    function isImageFile(file) {
      return !!(file && file.type && file.type.startsWith('image/'));
    }

    function loadGeminiApiKey() {
      try {
        geminiApiKey.value = localStorage.getItem(STORAGE_GEMINI_KEY) || '';
      } catch (_) {
        geminiApiKey.value = '';
      }
    }

    function saveGeminiApiKey() {
      try {
        localStorage.setItem(STORAGE_GEMINI_KEY, (geminiApiKey.value || '').trim());
        translateInfo.textContent = 'APIキーを保存しました。';
      } catch (_) {
        translateInfo.textContent = 'APIキー保存に失敗しました。';
      }
    }

    async function translateToJapaneseWithGemini(text) {
      const apiKey = (geminiApiKey.value || '').trim();
      if (!apiKey) throw new Error('Gemini API Keyを入力してください');

      const prompt = `次の文章を自然な日本語に翻訳してください。翻訳結果のみ返してください。\\n\\n${text}`;
      const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${encodeURIComponent(apiKey)}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          contents: [{ parts: [{ text: prompt }] }],
          generationConfig: { temperature: 0.2 }
        })
      });
      if (!res.ok) throw new Error(`翻訳APIエラー: ${res.status}`);
      const json = await res.json();
      const out = json?.candidates?.[0]?.content?.parts?.[0]?.text?.trim();
      if (!out) throw new Error('翻訳結果が空でした');
      return out;
    }

    async function translateSelectedRange() {
      const text = sourceText.value || '';
      if (!text.trim()) {
        translateInfo.textContent = '原文テキストを入力してください。';
        return;
      }
      const start = sourceText.selectionStart ?? 0;
      const end = sourceText.selectionEnd ?? 0;
      const selected = start !== end ? text.slice(start, end) : text;

      btnTranslateSelected.disabled = true;
      translateInfo.textContent = start !== end ? `選択範囲（${end - start}文字）を翻訳中...` : '全文を翻訳中...';
      try {
        const ja = await translateToJapaneseWithGemini(selected);
        translatedText.value = ja;
        translateInfo.textContent = '翻訳完了。必要に応じて結果を直接編集してください。';
      } catch (e) {
        translateInfo.textContent = `翻訳失敗: ${e?.message || e}`;
      } finally {
        btnTranslateSelected.disabled = false;
      }
    }

    function clearDropUi() {
      windowDragCounter = 0;
      dropZone.classList.remove('dragover');
      globalDropOverlay.classList.remove('show');
    }

    function defaultSplitPoints() {
      splitPoints = [];
      if (!originalImage) return;
      for (let y = 1000; y < originalImage.height; y += 1000) splitPoints.push(y);
      normalizeSplitPoints();
    }

    function normalizeSplitPoints() {
      if (!originalImage) return;
      splitPoints = splitPoints
        .map((y) => Math.round(y))
        .filter((y) => Number.isFinite(y))
        .map((y) => Math.max(10, Math.min(originalImage.height - 10, y)))
        .sort((a, b) => a - b);
      const dedup = [];
      splitPoints.forEach((y) => {
        if (dedup.length === 0 || Math.abs(dedup[dedup.length - 1] - y) >= 3) dedup.push(y);
      });
      splitPoints = dedup;
    }

    function loadImage(file) {
      if (!isImageFile(file)) {
        alert('画像ファイルを選択してください。');
        return;
      }
      clearDropUi();
      const url = URL.createObjectURL(file);
      const img = new Image();
      img.onload = () => {
        if (objectUrl) URL.revokeObjectURL(objectUrl);
        objectUrl = url;
        originalImage = img;
        previewImage.src = objectUrl;
        fileNameEl.textContent = file.name;
        fileInfo.style.display = 'flex';
        previewContainer.style.display = 'block';
        previewEmpty.style.display = 'none';
        btnProcess.disabled = false;
        defaultSplitPoints();
        applyZoom(zoomPercent);
        renderCutLines();
        updateManualList();
      };
      img.onerror = () => {
        URL.revokeObjectURL(url);
        alert('画像の読み込みに失敗しました。');
      };
      img.src = url;
    }

    function syncOverlaySize() {
      if (!originalImage) return;
      const rect = previewImage.getBoundingClientRect();
      if (!rect.width || !rect.height) return;
      const w = Math.round(rect.width);
      const h = Math.round(rect.height);
      previewStage.style.width = `${w}px`;
      previewStage.style.height = `${h}px`;
      overlayCanvas.width = w;
      overlayCanvas.height = h;
      overlayCanvas.style.width = `${w}px`;
      overlayCanvas.style.height = `${h}px`;
      cutLinesLayer.style.width = `${w}px`;
      cutLinesLayer.style.height = `${h}px`;
      drawLineGuides();
    }

    function drawLineGuides() {
      if (!originalImage) return;
      const rect = previewImage.getBoundingClientRect();
      if (!rect.height) return;
      octx.clearRect(0, 0, overlayCanvas.width, overlayCanvas.height);
      const scale = rect.height / originalImage.height;
      splitPoints.forEach((y, idx) => {
        const dy = Math.round(y * scale);
        octx.beginPath();
        octx.moveTo(0, dy);
        octx.lineTo(overlayCanvas.width, dy);
        octx.lineWidth = idx === selectedLineIndex ? 2 : 1;
        octx.strokeStyle = idx === selectedLineIndex ? 'rgba(56,189,248,.5)' : 'rgba(239,68,68,.32)';
        octx.setLineDash([5, 5]);
        octx.stroke();
        octx.setLineDash([]);
      });
    }

    function renderCutLines() {
      cutLinesLayer.innerHTML = '';
      if (!originalImage) return;
      const rect = previewImage.getBoundingClientRect();
      if (!rect.height) return;
      const scale = rect.height / originalImage.height;
      splitPoints.forEach((y, idx) => {
        const line = document.createElement('div');
        line.className = 'cut-line';
        if (idx === selectedLineIndex) line.classList.add('selected');
        line.style.top = `${Math.round(y * scale)}px`;
        line.dataset.index = String(idx);
        line.addEventListener('mousedown', (e) => {
          e.preventDefault();
          e.stopPropagation();
          isDraggingLine = true;
          draggingIndex = idx;
          dragLastClientY = e.clientY;
          selectedLineIndex = idx;
          renderCutLines();
          updateManualList();
        });
        line.addEventListener('click', (e) => {
          e.stopPropagation();
          selectedLineIndex = idx;
          renderCutLines();
          updateManualList();
        });
        cutLinesLayer.appendChild(line);
      });
      drawLineGuides();
    }

    function updateDraggingLineByClientY(clientY) {
      if (!isDraggingLine || draggingIndex < 0 || !originalImage) return;
      const r = previewImage.getBoundingClientRect();
      if (!r.height) return;
      let y = Math.round((clientY - r.top) * (originalImage.height / r.height));
      y = Math.max(10, Math.min(originalImage.height - 10, y));
      splitPoints[draggingIndex] = y;
      normalizeSplitPoints();
      renderCutLines();
    }

    function stopInnerDragScroll() {
      if (dragScrollTimer) {
        clearInterval(dragScrollTimer);
        dragScrollTimer = null;
      }
      dragScrollDir = 0;
    }

    function startInnerDragScroll(dir) {
      if (dragScrollTimer && dragScrollDir === dir) return;
      stopInnerDragScroll();
      dragScrollDir = dir;
      dragScrollTimer = setInterval(() => {
        if (!isDraggingLine) {
          stopInnerDragScroll();
          return;
        }
        const before = previewWrapper.scrollTop;
        previewWrapper.scrollTop += dir * 22;
        const moved = previewWrapper.scrollTop !== before;
        if (!moved) return;
        // マウスを押したままでも、内側スクロールに追従して線を更新
        updateDraggingLineByClientY(dragLastClientY);
      }, 16);
    }

    function updateManualList() {
      manualList.innerHTML = '';
      normalizeSplitPoints();
      splitPoints.forEach((pt, idx) => {
        const item = document.createElement('div');
        item.className = 'manual-item';
        if (idx === selectedLineIndex) item.style.background = 'rgba(56,189,248,.16)';
        item.innerHTML = `
          <span style="width:30px; color:#cbd5e1;">#${idx + 1}</span>
          <input type="number" value="${pt}">
          <span style="color:#94a3b8;">px</span>
          <button class="btn" style="padding:2px 8px; margin-left:auto;">×</button>
        `;
        const input = item.querySelector('input');
        const removeBtn = item.querySelector('button');
        input.addEventListener('change', () => {
          const v = parseInt(input.value, 10);
          if (!Number.isNaN(v) && originalImage) {
            splitPoints[idx] = Math.max(10, Math.min(originalImage.height - 10, v));
            normalizeSplitPoints();
            renderCutLines();
            updateManualList();
          }
        });
        removeBtn.addEventListener('click', () => {
          splitPoints.splice(idx, 1);
          selectedLineIndex = -1;
          normalizeSplitPoints();
          renderCutLines();
          updateManualList();
        });
        item.addEventListener('click', (e) => {
          if (e.target === input || e.target === removeBtn) return;
          selectedLineIndex = idx;
          renderCutLines();
          updateManualList();
        });
        manualList.appendChild(item);
      });
    }

    function addCutLineAtClientY(clientY) {
      if (!originalImage) return;
      const rect = previewImage.getBoundingClientRect();
      if (!rect.height) return;
      const offsetY = clientY - rect.top;
      if (offsetY < 0 || offsetY > rect.height) return;
      const y = Math.round(offsetY * (originalImage.height / rect.height));
      const clamped = Math.max(10, Math.min(originalImage.height - 10, y));
      splitPoints.push(clamped);
      normalizeSplitPoints();
      selectedLineIndex = splitPoints.findIndex((v) => v === clamped);
      if (selectedLineIndex < 0) selectedLineIndex = splitPoints.length - 1;
      renderCutLines();
      updateManualList();
    }

    function getEffectiveCutPolicy() {
      if (cutPolicy !== 'auto') return cutPolicy;
      if (!originalImage) return 'fit';
      return (originalImage.height / originalImage.width) > 1.6 ? 'scroll' : 'fit';
    }

    function getBaseDisplayWidth() {
      const containerWidth = Math.max(200, previewWrapper.clientWidth || previewWrapper.parentElement.clientWidth);
      if (!originalImage) return containerWidth;
      const effectivePolicy = getEffectiveCutPolicy();
      if (effectivePolicy === 'fit') {
        const wr = previewWrapper.getBoundingClientRect();
        const maxHeight = Math.max(260, window.innerHeight - wr.top - 24);
        const fitByHeight = maxHeight / (originalImage.height / originalImage.width);
        return Math.min(containerWidth, fitByHeight);
      }
      // scroll: 横幅はコンテナ幅に合わせて、必要ならズームで拡大
      return containerWidth;
    }

    function setCutPolicy(policy) {
      cutPolicy = policy;
      applyZoom(zoomPercent);
    }

    function applyZoom(percent) {
      zoomPercent = Math.max(20, Math.min(300, parseInt(percent, 10) || 100));
      zoomSlider.value = String(zoomPercent);
      zoomValue.textContent = `${zoomPercent}%`;
      if (!originalImage) return;

      const oldWidth = previewImage.getBoundingClientRect().width || 1;
      const oldCenterX = previewWrapper.scrollLeft + (previewWrapper.clientWidth / 2);
      const centerRatio = oldCenterX / oldWidth;

      const baseWidth = getBaseDisplayWidth();
      const displayWidth = Math.round(baseWidth * (zoomPercent / 100));
      previewImage.style.width = `${displayWidth}px`;
      previewImage.style.maxWidth = 'none';
      previewStage.style.width = `${displayWidth}px`;
      previewStage.style.minWidth = `${displayWidth}px`;
      syncOverlaySize();
      renderCutLines();

      // 拡大縮小で右にズレる問題の対策: ビューポート中心を維持
      requestAnimationFrame(() => {
        const newWidth = previewImage.getBoundingClientRect().width || displayWidth;
        if (newWidth <= previewWrapper.clientWidth) {
          previewWrapper.scrollLeft = 0;
          return;
        }
        const target = (centerRatio * newWidth) - (previewWrapper.clientWidth / 2);
        previewWrapper.scrollLeft = Math.max(0, Math.min(target, newWidth - previewWrapper.clientWidth));
      });
    }

    function processCuts() {
      if (!originalImage) return;
      normalizeSplitPoints();
      const points = [0, ...splitPoints, originalImage.height];
      const baseName = (fileNameEl.textContent || 'image').replace(/\.[^/.]+$/, '');
      results.innerHTML = '';
      resultImages = [];

      for (let i = 0; i < points.length - 1; i += 1) {
        const y = points[i];
        const h = points[i + 1] - y;
        if (h < 5) continue;
        const canvas = document.createElement('canvas');
        canvas.width = originalImage.width;
        canvas.height = h;
        const ctx = canvas.getContext('2d');
        ctx.fillStyle = '#fff';
        ctx.fillRect(0, 0, canvas.width, canvas.height);
        ctx.drawImage(originalImage, 0, y, originalImage.width, h, 0, 0, originalImage.width, h);

        const url = canvas.toDataURL('image/jpeg', 0.92);
        const filename = `${baseName}_${String(i + 1).padStart(2, '0')}.jpg`;
        resultImages.push({ url, filename });

        const card = document.createElement('div');
        card.className = 'result';
        card.innerHTML = `<img src="${url}" alt="cut-${i + 1}"><button class="btn btn-primary" style="width:100%;">保存</button>`;
        card.querySelector('button').addEventListener('click', () => downloadOne(url, filename));
        results.appendChild(card);
      }

      btnDownloadAll.style.display = resultImages.length ? 'block' : 'none';
    }

    function downloadOne(url, filename) {
      const a = document.createElement('a');
      a.href = url;
      a.download = filename;
      a.click();
    }

    async function downloadAll() {
      for (const item of resultImages) {
        downloadOne(item.url, item.filename);
        await new Promise((r) => setTimeout(r, 180));
      }
    }

    function resetLines() {
      if (!originalImage) return;
      defaultSplitPoints();
      selectedLineIndex = -1;
      renderCutLines();
      updateManualList();
    }

    function setupDnD() {
      ['dragenter', 'dragover', 'dragleave', 'drop'].forEach((ev) => {
        window.addEventListener(ev, (e) => {
          // 常に既定動作を止めて、ブラウザのタブ遷移を防ぐ
          e.preventDefault();
          e.stopPropagation();
        }, { passive: false });
      });
      window.addEventListener('dragenter', (e) => {
        if (!hasFiles(e)) return;
        windowDragCounter += 1;
        globalDropOverlay.classList.add('show');
      }, { passive: false });
      window.addEventListener('dragover', (e) => {
        if (!hasFiles(e)) return;
        globalDropOverlay.classList.add('show');
      }, { passive: false });
      window.addEventListener('dragleave', () => {
        windowDragCounter = Math.max(0, windowDragCounter - 1);
        if (windowDragCounter === 0) clearDropUi();
      }, { passive: false });
      window.addEventListener('drop', (e) => {
        e.preventDefault();
        e.stopPropagation();
        clearDropUi();
        const file = e.dataTransfer.files && e.dataTransfer.files[0];
        if (file) loadImage(file);
      }, { passive: false });
      window.addEventListener('dragend', clearDropUi, { passive: false });
      window.addEventListener('blur', clearDropUi);
    }

    function hasFiles(e) {
      try {
        return e.dataTransfer && Array.from(e.dataTransfer.types || []).includes('Files');
      } catch (_) {
        return false;
      }
    }

    function bindEvents() {
      btnSaveApiKey.addEventListener('click', saveGeminiApiKey);
      btnTranslateSelected.addEventListener('click', translateSelectedRange);
      dropZone.addEventListener('click', () => fileInput.click());
      dropZone.addEventListener('keydown', (e) => {
        if (e.key === 'Enter' || e.key === ' ') {
          e.preventDefault();
          fileInput.click();
        }
      });
      dropZone.addEventListener('dragover', (e) => {
        e.preventDefault();
        dropZone.classList.add('dragover');
      });
      dropZone.addEventListener('dragleave', (e) => {
        if (!dropZone.contains(e.relatedTarget)) dropZone.classList.remove('dragover');
      });
      dropZone.addEventListener('drop', (e) => {
        e.preventDefault();
        e.stopPropagation();
        dropZone.classList.remove('dragover');
        const file = e.dataTransfer.files && e.dataTransfer.files[0];
        if (file) loadImage(file);
      });
      fileInput.addEventListener('change', (e) => {
        const file = e.target.files && e.target.files[0];
        if (file) loadImage(file);
        fileInput.value = '';
      });
      // 透明inputにドラッグした場合もハイライト解除を保証
      fileInput.addEventListener('dragenter', () => dropZone.classList.add('dragover'));
      fileInput.addEventListener('dragover', () => dropZone.classList.add('dragover'));
      fileInput.addEventListener('dragleave', () => dropZone.classList.remove('dragover'));
      fileInput.addEventListener('drop', () => clearDropUi());
      if (btnChangeFile) {
        btnChangeFile.addEventListener('click', () => fileInput.click());
      }

      previewImage.addEventListener('load', () => {
        syncOverlaySize();
        renderCutLines();
      });

      previewWrapper.addEventListener('dblclick', (e) => {
        // ダブルクリック追加を一箇所に統一
        if (!originalImage) return;
        if (e.target.closest('.cut-line')) return;
        addCutLineAtClientY(e.clientY);
      });

      document.addEventListener('mousemove', (e) => {
        if (!isDraggingLine || draggingIndex < 0 || !originalImage) return;
        e.preventDefault();
        dragLastClientY = e.clientY;

        // 端に来たら「内側プレビュー」だけ自動スクロール
        const wr = previewWrapper.getBoundingClientRect();
        const edge = 50;
        const yInWrapper = e.clientY - wr.top;
        if (yInWrapper >= 0 && yInWrapper < edge) {
          startInnerDragScroll(-1);
        } else if (yInWrapper <= wr.height && yInWrapper > wr.height - edge) {
          startInnerDragScroll(1);
        } else {
          stopInnerDragScroll();
        }

        updateDraggingLineByClientY(dragLastClientY);
      });

      document.addEventListener('mouseup', () => {
        if (!isDraggingLine) return;
        isDraggingLine = false;
        draggingIndex = -1;
        stopInnerDragScroll();
        renderCutLines();
        updateManualList();
      });

      document.addEventListener('keydown', (e) => {
        if ((e.key === 'Delete' || e.key === 'Backspace') && selectedLineIndex !== -1) {
          e.preventDefault();
          splitPoints.splice(selectedLineIndex, 1);
          selectedLineIndex = -1;
          normalizeSplitPoints();
          renderCutLines();
          updateManualList();
        }
      });

      zoomSlider.addEventListener('input', () => applyZoom(zoomSlider.value));
      document.querySelectorAll('input[name="cutPolicy"]').forEach((r) => {
        r.addEventListener('change', () => setCutPolicy(r.value));
      });
      btnProcess.addEventListener('click', processCuts);
      btnDownloadAll.addEventListener('click', downloadAll);
      btnReset.addEventListener('click', resetLines);
      window.addEventListener('resize', () => {
        applyZoom(zoomPercent);
        syncOverlaySize();
      });
    }

    loadGeminiApiKey();
    setupDnD();
    bindEvents();
  </script>
</body>
</html>
